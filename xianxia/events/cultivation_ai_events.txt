# Yearly path-opinion normalization: same path (even cross-sect) gets goodwill, different paths get rivalry.
cultivation_ai.2280 = {
    type = character_event
    scope = none
    hidden = yes

    immediate = {
        # Performance-safe: normalize only among independent rulers in diplomatic reach.
        every_independent_ruler = {
            limit = {
                is_alive = yes
                is_adult = yes
                is_cultivator = yes
                has_cultivation_sect_faith = yes
            }

            every_independent_ruler = {
                limit = {
                    is_alive = yes
                    is_adult = yes
                    is_cultivator = yes
                    has_cultivation_sect_faith = yes
                    NOT = { this = scope:root }
                    is_within_diplo_range = scope:root
                }

                # Clear stale path-opinion state before applying current relationship.
                remove_opinion = { target = root modifier = opinion_cultivation_path_peer }
                remove_opinion = { target = root modifier = opinion_cultivation_path_rival }

                if = {
                    limit = {
                        OR = {
                            AND = {
                                faith = { has_doctrine_parameter = orthodox_cultivation }
                                root.faith = { has_doctrine_parameter = orthodox_cultivation }
                            }
                            AND = {
                                faith = { has_doctrine_parameter = unorthodox_cultivation }
                                root.faith = { has_doctrine_parameter = unorthodox_cultivation }
                            }
                            AND = {
                                faith = { has_doctrine_parameter = demonic_cultivation }
                                root.faith = { has_doctrine_parameter = demonic_cultivation }
                            }
                            AND = {
                                faith = { has_doctrine_parameter = vagrant_cultivation }
                                root.faith = { has_doctrine_parameter = vagrant_cultivation }
                            }
                        }
                    }
                    add_opinion = { target = root modifier = opinion_cultivation_path_peer years = 20 }
                }
                else = {
                    add_opinion = { target = root modifier = opinion_cultivation_path_rival years = 20 }
                }
            }
        }
    }

    option = { name = OK }
}

        NOT = { has_character_flag = ai_recently_joined_sect }

        if = {
            limit = {
                NOT = { has_cultivation_sect_faith = yes }
            }

            cultivation_ai_join_any_sect_by_meridian_effect = yes

            add_character_flag = { flag = ai_recently_joined_sect days = 1825 }
        }
    }

    option = { name = OK }
}
