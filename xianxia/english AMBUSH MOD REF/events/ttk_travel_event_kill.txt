				factor = 0.10
				is_close_family_of = scope:ttk_target
			}
		}
	}

		name = ttk_travel_event_kill.0002.b

			limit = { 
				scope:ttk_target = { is_ai = yes } 
				scope:ttk_killer = { is_ai = no }
