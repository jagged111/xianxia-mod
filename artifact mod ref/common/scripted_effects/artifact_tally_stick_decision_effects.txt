artifact_tally_create_effect = {
	$OWNER$ = { save_scope_as = owner }
	
	hidden_effect = {
		random_county_in_region = {
			region = world_africa_sahara
			random_county_province = {
				save_scope_as = tally_found_here
			}
		}
	}

	hidden_effect_new_object = {
		scope:owner = {
			create_artifact = {
				name = artifact_tally_name
				description = artifact_tally_desc
				rarity = illustrious
				visuals = pedestal_branch_relic_germanic
				type = pedestal
				template = artifact_tally_template

				history = {
					location = scope:tally_found_here
					actor = root
					recipient = root
					type = discovered
				}
				decaying = no
				
				quality = 1000
				wealth = 1000

				save_scope_as = newly_created_artifact
				modifier = artifact_court_grandeur_baseline_add_6_modifier
				modifier = artifact_tally_modifier
			}
		}
	}

	#scope:newly_created_artifact = {
		#get_artifact_feature_references_effect = yes
		#set_artifact_feature_group = decoration_material_wire
		#set_artifact_feature_group = decoration_material_gem
	#}
} # end artifact_tally_create_effect

artifact_tally_teach_me_effect = {
	scope:the_artifact_user = {
		random_list = {
			50 = {
				trigger = {
					age >= 16
					NOT = { has_trait = education_stewardship }
				}
				add_trait = education_stewardship_1
			}
			40 = {
				trigger = {
					age >= 16
					AND = {
						has_trait = education_stewardship
						NOT = { has_trait = education_stewardship_5 }
					}
				}
				change_trait_rank = {
					trait = education_stewardship
					rank = 1
					max = 5
				}
			}
			30 = {
				trigger = {
					age >= 16
					NOT = { has_trait = education_republican_knowledge }
				}
				add_trait = education_republican_knowledge_1
			}
			30 = {
				trigger = {
					age = 16
					NOT = { has_trait = education_republican_knowledge_4 }
					has_trait = education_republican_knowledge
				}
			
				change_trait_rank = {
					trait = education_republican_knowledge
					rank = 1
					max = 4
				}
			}
			10 = { 
				trigger = { 
					NOT = { has_trait = administrative_court_2 }
					NOT = { has_trait = administrative_court_1 }
				}
				add_trait = administrative_court_1					
			}
			10 = { 
				trigger = { has_trait = administrative_court_1 }
				change_trait_rank = {
					trait = administrative_court
					rank = 1
					max = 2
				}					
			}
			20 = { 
				trigger = { stewardship <= 99 }
				send_interface_toast = {
					title = artifact_use_tally_decision
					scope:the_artifact_user = {
						add_stewardship_skill = 10
					}
				}					
			}					
			20 = {
				trigger = {
					OR = {
						NOT = { has_perk = avaricious_perk }
						NOT = { has_perk = architect_perk }
						NOT = { has_perk = administrator_perk }
					}
				}
				send_interface_toast = {
					title = artifact_use_tally_decision
					scope:the_artifact_user = {
						add_stewardship_lifestyle_perk_points = 1
					}
				}
			}
		} # end random_list
		force_character_skill_recalculation = yes
	} # end scope:artifact_user
} # end artifact_tally_teach_me_effect

artifact_tally_improve_steward_effect = {
	custom_tooltip = artifact_tally_improve_steward_effect_tt
	
	scope:the_artifact_user = {
		# get steward
		cp:councillor_steward = {
			random_list = {
				500 = {
					trigger = {
						NOT = { has_trait = education_stewardship }
					}
					scope:the_artifact_user = {
						send_interface_toast = {
							title = artifact_use_tally_decision
							left_icon = cp:councillor_steward
							cp:councillor_steward = {
								add_trait = education_stewardship_1
							}
						} # end send_interface_toast
					}
				}					
				500 = {
					trigger = {
						has_trait = education_stewardship
						NOT = { has_trait = education_stewardship_5 }
					}
					scope:the_artifact_user = {
						send_interface_toast = {
							title = artifact_use_tally_decision
							left_icon = cp:councillor_steward
							cp:councillor_steward = {
								change_trait_rank = {
									trait = education_stewardship
									rank = 1
									max = 5
								}
							}
						} # end send_interface_toast
					}
				}
				10 = { 
					trigger = { NOT = { has_trait = administrative_court_2 }}
					add_trait_force_tooltip = administrative_court_2 
					hidden_effect = { remove_trait = administrative_court_1 }				
					scope:the_artifact_user = {
						send_interface_toast = {
							title = artifact_use_tally_decision
							left_icon = cp:councillor_steward
							cp:councillor_steward = {
								add_trait = administrative_court_2
							}
						}
					}
				}
				1000 = { # Almost always pick this, or one of the education traits 
					trigger = { stewardship <= 99 }
					scope:the_artifact_user = {
						send_interface_toast = {
							title = artifact_use_tally_decision
							left_icon = cp:councillor_steward
							cp:councillor_steward = {
								add_stewardship_skill = 10 
							}
						}
					}
				}
			} # end random_list
			force_character_skill_recalculation = yes
		} # end cp:councillor_steward
	}
} # end artifact_tally_improve_steward_effect

artifact_tally_recruit_seneschal_effect = {
	custom_tooltip = artifact_tally_recruit_seneschal_effect_tooltip

	hidden_effect = {
		scope:the_artifact_user = {
			every_sub_realm_county = {
				this.culture = {
					add_to_list = nearby_cultures	
				}
			}
			every_character_to_title_neighboring_and_across_water_county = {
				this.culture = {
					add_to_list = nearby_cultures
				}
			}
		} # end scope:the_artifact_user
		every_culture_global = {
			limit = {
				OR = {
					has_cultural_parameter = next_level_trade_ports						
				}
			}
			add_to_list = nearby_cultures
		}

		culture:greek = { add_to_list = nearby_cultures	}
			
		random_in_list = {
			list = nearby_cultures
			save_scope_as = the_recruit_culture
		}
		
		# clear the list
		clear_variable_list = nearby_cultures

		create_character = {
			#location = scope:the_artifact_user.capital_province
			template = artifact_tally_seneschal		
			save_scope_as = the_recruit
			employer = scope:the_artifact_user
			dynasty = none
		} # End create character
		
		scope:the_recruit = {
			set_relation_friend = scope:the_artifact_user
		}
	} # end hidden_effect

	send_interface_toast = {
		type = character_arrived
		title = msg_artifact_tally_recruit_seneschal_effect_title
		desc = msg_artifact_tally_recruit_seneschal_effect_desc
		left_icon = scope:the_recruit
	}

} # end artifact_tally_recruit_seneschal_effect

artifact_tally_add_development_effect = {
	scope:the_artifact_user = {
		every_sub_realm_county = {
			if = {
				limit = { 
					holder = scope:the_artifact_user
					development_level <= 99
				}
				change_development_progress_with_overflow = 50
			}
			else_if = {
				limit = { 
					development_level <= 99
				}
				change_development_progress_with_overflow = 25				
			}
		}
	}
} # end artifact_tally_add_development_effect
