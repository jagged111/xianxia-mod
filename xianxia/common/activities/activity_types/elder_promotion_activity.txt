activity_elder_promotion = {
	is_shown = {
		is_adult = yes
		is_landed = yes
		is_cultivator = yes
	}

	can_start_showing_failures_only = {
		NOT = { is_activity_type_on_cooldown = activity_elder_promotion }
		is_available_adult = yes
		is_landed = yes
		is_cultivator = yes
	}

	is_valid = {
		scope:host = {
			is_alive = yes
			is_capable_adult = yes
			is_imprisoned = no
			is_landed = yes
			is_cultivator = yes
				any_courtier = {
					is_alive = yes
					is_adult = yes
					OR = {
						has_character_modifier = inner_court_disciple
						has_trait = sect_disciple
					}
					has_elder_eligible_realm = yes
					NOT = { has_trait = sect_elder }
				}
			}
		}

	on_invalidated = {
		invalidation_imprisoned_effect = yes
		invalidation_incapable_effect = yes
	}

	province_filter = realm
	ai_province_filter = realm
	max_province_icons = 1
	allow_zero_guest_invites = yes
	open_invite = no
	cooldown = { years = 12 }

	can_be_activity_guest = {
		is_alive = yes
		is_adult = yes
		is_cultivator = yes
	}

	phases = {
		elder_promotion_phase_nominations = {
			is_predefined = yes
			location_source = pickable
			on_phase_active = {
				if = {
					limit = { this = scope:host }
					trigger_event = xianxia_activity.7500
					scope:activity = { progress_activity_phase_after = { days = 12 } }
				}
			}
		}

		elder_promotion_phase_assessment = {
			is_predefined = yes
			on_phase_active = {
				if = {
					limit = { this = scope:host }
					trigger_event = xianxia_activity.7510
					scope:activity = { progress_activity_phase_after = { days = 10 } }
				}
			}
		}

		elder_promotion_phase_enthronement = {
			is_predefined = yes
			on_phase_active = {
				if = {
					limit = { this = scope:host }
					trigger_event = xianxia_activity.7520
				}
			}
		}
	}
}
