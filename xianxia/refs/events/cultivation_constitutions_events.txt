#Dragon bloodline unlock by eating a dragon heart
cultivation_constitutions_events.002 = {

	type = character_event
	title = cultivation_constitutions_events.002.t
	desc = cultivation_constitutions_events.002.desc

	theme = widgetless_travel_danger 
	left_portrait = {
		character = root
		animation = thinking
	}

	trigger = {
		and = {
			not = {
				has_trait = dragon_bloodline
			}
			not = {
				has_trait = dragon_body_constitution
			}
			is_cultivator = yes
		}
	}

	option = { #You eat the heart
        name = cultivation_constitutions_events.002.a

		ai_chance = {
			base = 40
			modifier = {
				add = cultivation_attitude
			}
			modifier = {
				has_trait = ambitious
				add = 10
			}
		}
		random_list = {
			10 = {#Get the bloodline, no consequences
				add_trait = dragon_bloodline
			}
			40 = {#Get the bloodline but also get ill
				add_trait = dragon_bloodline
				add_trait = wounded_1
			}
			40 = {#Get the bloodline but also get ill and injured
				add_trait = dragon_bloodline
				add_trait = wounded_1
				wound_cultivation_character = yes
			}
			10 = {
				death = {
					death_reason = death_evolution
				}
			}
		}
	}

	option = {#You ignore it
        name = cultivation_constitutions_events.002.b

		ai_chance = {
			base = 50
			modifier = {
				has_trait = craven
				add = 10
			}
		}
	}
}
