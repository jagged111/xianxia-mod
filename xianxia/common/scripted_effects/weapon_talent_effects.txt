xianxia_apply_weapon_talent_lottery_effect = {
	remove_trait = weapon_combat_talent_1
	remove_trait = weapon_combat_talent_2
	remove_trait = weapon_combat_talent_3
	remove_trait = weapon_combat_talent_4
	remove_trait = weapon_combat_talent_5

	if = {
		limit = {
			OR = {
				has_trait = brave

			}
		}
		random_list = {
			24 = { add_trait = weapon_combat_talent_1 }

			10 = { add_trait = weapon_combat_talent_5 }
		}
	}

		limit = {
			OR = {
				has_trait = craven

			}
		}
