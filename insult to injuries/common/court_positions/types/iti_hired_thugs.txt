iti_thug_court_position = {
	sort_order = 250
	max_available_positions = 3
	skill = prowess

	court_position_asset = {
		animation = menacing
		background = "gfx/interface/illustrations/event_scenes/tavern.dds"
	}

	opinion = {
		value = regular_court_position_opinion
	}

	
	on_court_position_received = {
		basic_gained_court_position_effect = yes
		thug_title_accepted_effect = yes
		scope:employee = {
			add_character_flag = is_hired_thug
			add_character_flag = blocked_from_leaving
		}
	}
	
	on_court_position_revoked = {
		basic_revoked_court_position_effect = yes
		thug_title_revoked_effect = yes
		scope:employee = {
			remove_character_flag = is_hired_thug
			remove_character_flag = blocked_from_leaving
		}
	}
	on_court_position_invalidated = {
		basic_invalidated_court_position_effect = yes
		thug_title_invalidated_effect = yes
		scope:employee = {
			remove_character_flag = is_hired_thug
			remove_character_flag = blocked_from_leaving
		}
	}

	aptitude_level_breakpoints = { 20 40 60 80 }

	aptitude = {
		value = 1
		add = court_position_aptitude_boon_voyager_trait
		
		# Adventurer Perks
		if = {
			limit = {
				employer ?= {
					government_has_flag = government_is_landless_adventurer
					has_perk = soon_forgiven_perk
				}
			}
			add = { value = 10 desc = court_position_soon_forgiven_perk }
		}

		# Skills
		add = {
			value = prowess
			multiply = 4
			max = 50
			desc = court_position_skill_prowess
		}
		add = {
			value = intrigue
			multiply = 1.25
			max = 50
			desc = court_position_skill_intrigue
		}
		
		# Personality & Standard Traits
		if = {
			limit = { has_trait = brave }
			add = { value = 25 desc = court_position_brave_trait }
		}
		if = {
			limit = { has_trait = wrathful }
			add = { value = 20 desc = court_position_wrathful_trait }
		}
		if = {
			limit = { has_trait = sadistic }
			add = { value = 25 desc = court_position_sadistic_trait }
		}
		if = {
			limit = { has_trait = callous }
			add = { value = 20 desc = court_position_callous_trait }
		}
		if = {
			limit = { has_trait = vengeful }
			add = { value = 15 desc = court_position_vengeful_trait }
		}
		if = {
			limit = { has_trait = arbitrary }
			add = { value = 20 desc = court_position_arbitrary_trait }
		}
		if = {
			limit = { has_trait = deceitful }
			add = { value = 15 desc = court_position_deceitful_trait }
		}

		# Negative Traits
		if = {
			limit = { has_trait = craven }
			add = { value = -30 desc = court_position_craven_trait }
		}
		if = {
			limit = { has_trait = calm }
			add = { value = -15 desc = court_position_calm_trait }
		}
		if = {
			limit = { has_trait = compassionate }
			add = { value = -50 desc = court_position_compassionate_trait }
		}
		if = {
			limit = { has_trait = just }
			add = { value = -20 desc = court_position_just_trait }
		}
		if = {
			limit = { has_trait = honest }
			add = { value = -20 desc = court_position_honest_trait }
		}

		# Reputation & Lifestyle
		if = {
			limit = { has_trait = murderer }
			add = { value = 20 desc = court_position_murderer_trait }
		}
		if = {
			limit = { has_trait = torturer }
			add = { value = 20 desc = court_position_torturer_trait }
		}

		# Physique Traits (Big and Strong)
		if = {
			limit = { has_trait = giant }
			add = { value = 20 desc = court_position_giant_trait }
		}
		if = {
			limit = { has_trait = strong }
			add = { value = 15 desc = court_position_strong_trait }
		}
		if = {
			limit = { has_trait = physique_good_3 }
			add = { value = 20 desc = court_position_physique_good_3_trait }
		}
		else_if = {
			limit = { has_trait = physique_good_2 }
			add = { value = 15 desc = court_position_physique_good_2_trait }
		}
		else_if = {
			limit = { has_trait = physique_good_1 }
			add = { value = 10 desc = court_position_physique_good_1_trait }
		}

		# Beauty Traits (Ugly and Hideous)
		if = {
			limit = { has_trait = beauty_bad_3 }
			add = { value = 20 desc = court_position_beauty_bad_3_trait }
		}
		else_if = {
			limit = { has_trait = beauty_bad_2 }
			add = { value = 15 desc = court_position_beauty_bad_2_trait }
		}
		else_if = {
			limit = { has_trait = beauty_bad_1 }
			add = { value = 10 desc = court_position_beauty_bad_1_trait }
		}

		# Thuggery & Intimidation Traits
		if = {
			limit = { has_trait = gallowsbait }
			add = { value = 20 desc = court_position_gallowsbait_trait }
		}
		if = {
			limit = { has_trait = scarred }
			add = { value = 10 desc = court_position_scarred_trait }
		}
		if = {
			limit = { has_trait = disfigured }
			add = { value = 15 desc = court_position_disfigured_trait }
		}
		if = {
			limit = { has_trait = one_eyed }
			add = { value = 10 desc = court_position_one_eyed_trait }
		}
		if = {
			limit = { has_trait = scaly }
			add = { value = 15 desc = court_position_scaly_trait }
		}
		if = {
			limit = {
				OR = {
					has_trait = lunatic_1
					has_trait = possessed_1
				}
			}
			add = { value = 15 desc = court_position_lunatic_trait }
		}
		
		# Relationship Bonuses
		if = {
			limit = { employer ?= { has_relation_soulmate = prev } }
			add = { value = 40 desc = court_position_soulmate }
		}
		else_if = {
			limit = { employer ?= { has_relation_best_friend = prev } }
			add = { value = 30 desc = court_position_best_friend }
		}
		else_if = {
			limit = { employer ?= { has_relation_friend = prev } }
			add = { value = 15 desc = court_position_friend }
		}

		add = court_position_aptitude_family_business_value
	}

	is_shown = {
		has_ep1_court_positions_dlc_trigger = yes
		trigger_if = {
			limit = {
				is_ai = yes
				highest_held_title_tier <= tier_county
			}
			always = no
		}
	}

	valid_character = {
		scope:employee = {
			is_adult = yes
			is_lowborn = yes
			is_ruler = no
			is_physically_able = yes
			prowess >= 5
		}
	}

	revoke_cost = {
		prestige = {
			value = 0 # No cost to fire
		}
	}

	salary = {
		gold = {
			value = 0
			if = {
				limit = { scope:liege = { has_treasury = no } }
				add = bodyguard_total_salary_value
			}
		}
		treasury = {
			value = 0
			if = {
				limit = { scope:liege = { has_treasury = yes } }
				add = bodyguard_total_salary_value
			}
		}
		prestige = {
			value = major_court_position_prestige_salary
		}
		round = no
	}

	received_salary = {
		gold = { value = bodyguard_total_salary_value }
		prestige = { value = major_court_position_prestige_salary }
		round = no
	}

	is_powerful_agent = no

	scaling_employer_modifiers = {
		terrible = { hostile_scheme_phase_duration_add = -1 enemy_hostile_scheme_success_chance_add = -2 }
		poor = { hostile_scheme_phase_duration_add = -2 enemy_hostile_scheme_success_chance_add = -4 }
		average = { hostile_scheme_phase_duration_add = -3 enemy_hostile_scheme_success_chance_add = -6 }
		good = { hostile_scheme_phase_duration_add = -4 enemy_hostile_scheme_success_chance_add = -8 }
		excellent = { hostile_scheme_phase_duration_add = -5 enemy_hostile_scheme_success_chance_add = -10 }
	}

	modifier = {
		owned_hostile_scheme_success_chance_add = 20
		dread_baseline_add = 5
	}

	ai_position_score = {
		value = 50
		scope:liege = {
			if = {
				limit = { ai_boldness >= 1 }
				add = 150
			}
			if = {
				limit = { has_trait = paranoid }
				add = 50
			}
		}
		add = court_position_debt_considerations_value
	}

	ai_candidate_score = {
		value = 50
		add = scope:base_value
		add = court_position_candidate_score_base_value
		add = court_position_candidate_aptitude_value
	}
}
