#clean event ## theme done
ttk_travel_event_duel.0001 = {
	type = character_event
	theme = rival_relation
	title = ttk_travel_event_duel.0001.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					scope:ttk_duel_init = {
						has_relation_nemesis = scope:ttk_duel_target
					}
				}
				desc = ttk_travel_event_duel.0001.nemesis
			}
			triggered_desc = {
				trigger = {
					scope:ttk_duel_init = {
						has_relation_rival = scope:ttk_duel_target
					}
				}
				desc = ttk_travel_event_duel.0001.rival
			}
		}
	}
	override_background = { reference = terrain_scope }
	left_portrait = {
		character = scope:ttk_duel_init
		animation = survey
		camera = camera_event_very_left
		}
	right_portrait = {
		character = scope:ttk_duel_target
		animation = jockey_walk 
		camera = camera_event_horse_right
		}
	trigger = {
		trigger_if = { 
			limit = {
				scope:ttk_duel_target = { is_travel_leader = no }
			}
			scope:ttk_duel_target.current_travel_plan.travel_plan_owner = {
				trigger_if = {
					limit  = { exists = involved_activity }
					NOT = { scope:ttk_duel_target.location = involved_activity.activity_location }
				}
				trigger_else = {
					always = yes
				}
			}
		}
		trigger_else = {
			scope:ttk_duel_target = {
				trigger_if = {
					limit = { exists = involved_activity }
					NOT = { scope:ttk_duel_target.location = involved_activity.activity_location }
				}
				trigger_else = {
					always = yes
				}
			}
		}		
	}			
	immediate = {
		if = {
			limit = {
				scope:ttk_duel_target = { is_travel_leader = no }
			}
			scope:ttk_duel_target = {
				current_travel_plan.travel_plan_owner = { save_scope_as = ttk_duel_target_tl }
			}
		}
		hidden_effect = {
			scope:ttk_duel_init = {
				add_character_flag = ttk_duel_can_pass
				current_travel_plan ?= { delay_travel_plan = { days = 3} }
			}
			scope:ttk_duel_target = {
				current_travel_plan ?= { delay_travel_plan = { days = 3} }
			}
			scope:ttk_duel_target_tl ?= {
				current_travel_plan ?= { delay_travel_plan = { days = 3} }
			}
		}
	}		
	option = {
		name = ttk_travel_event_duel.0001.a
		trigger_event = {
			id = ttk_travel_event_duel.0020
			days = 2
		}
	}
	option = {
		name = ttk_travel_event_duel.0001.b
		hidden_effect = {
			trigger_event = ttk_travel_event_duel.0008
		}
	}
}

ttk_travel_event_duel.0004 = {
	type = character_event
	theme = rival_relation
	title = ttk_travel_event_duel.0004.t
	desc = ttk_travel_event_duel.0004.desc
	override_background = { reference = terrain_scope }		
	left_portrait = {
		character = scope:ttk_duel_init
		scripted_animation = duel_celebrate   
		}
	right_portrait = {
		character = scope:ttk_duel_target
		animation = severelywounded 
		}
	immediate = {
		scope:ttk_duel_target = {
			house ?= {
				change_house_relation_effect = {
					HOUSE = scope:ttk_duel_init.house
					VALUE = house_relation_damage_minor_value
					REASON = duel
					CHAR = scope:ttk_duel_init
					TARGET_CHAR = scope:ttk_duel_target
					TITLE = scope:dummy_gender
				}
			}
		}
	}
	option = {
		name = ttk_travel_event_duel.0004.a
		ttk_execute_prisoner_effect = {
			VICTIM = scope:ttk_duel_target
			EXECUTIONER = scope:ttk_duel_init
		}
	}
	option = {
		name = ttk_travel_event_duel.0004.b
		scope:ttk_duel_init = {
			add_prestige = medium_prestige_gain
		}
		scope:ttk_duel_target = {
			add_prestige = medium_prestige_loss
		}
		if = {
			limit = {
				has_personality_benevolent_trigger = no
			}
			add_stress = medium_stress_gain
		}
		else_if = {
			limit = {
				has_personality_benevolent_trigger = yes
			}
			add_stress = medium_stress_loss
		}
	}
}
