namespace = iti_secrets

iti_secrets.1000 = {
	type = character_event
	title = iti_secrets.1000.t
	desc = iti_secrets.1000.desc
	theme = intrigue

	left_portrait = {
		character = root
		animation = disbelief
	}

	right_portrait = {
		character = scope:councillor
		animation = worried
	}

	lower_left_portrait = {
		character = scope:victim
		animation = head_bruised
	}

	immediate = {
		# Identify the Councillor
		random_councillor = {
			limit = {
				OR = {
					has_council_position = councillor_chancellor
					has_council_position = councillor_steward
					has_council_position = councillor_court_chaplain
					has_council_position = councillor_marshal
					has_council_position = councillor_spymaster
				}
			}
			save_scope_as = councillor
		}
		
		# Fallback if no councillor is found
		if = {
			limit = { NOT = { exists = scope:councillor } }
			random_courtier = {
				save_scope_as = councillor
			}
		}
	}

	option = {
		name = iti_secrets.1000.a
		add_stress = major_stress_impact_gain
	}
}
