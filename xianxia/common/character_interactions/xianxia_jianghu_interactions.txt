jianghu_face_slap_interaction = {
	category = interaction_category_hostile
	interface_priority = 120
	icon = icon_tyranny
	desc = jianghu_face_slap_interaction_desc
	use_diplomatic_range = yes

	ai_frequency = 36
	ai_targets = {
		ai_recipients = neighboring_rulers
		ai_recipients = peer_vassals
		max = 3
	}

	is_shown = {
		scope:recipient != scope:actor
		scope:actor = {
			is_alive = yes
			is_adult = yes
			is_cultivator = yes
		}
		scope:recipient = {
			is_alive = yes
			is_adult = yes
			OR = {
				is_cultivator = no
				has_trait = cultivation_qi_gathering
				has_trait = cultivation_qi_refining
				has_trait = cultivation_qi_building
				has_trait = cultivation_core_formation
			}
		}
	}

	is_valid_showing_failures_only = {
		scope:actor = {
			NOT = { has_character_flag = jianghu_face_slap_cooldown }
		}
	}

	auto_accept = yes

	ai_will_do = {
		base = 0
		modifier = { add = 55 has_relation_rival = scope:recipient }
		modifier = { add = 25 has_trait = arrogant }
		modifier = { add = 15 has_trait = wrathful }
		modifier = {
			factor = 0
			scope:actor = { opinion = { target = scope:recipient value > 15 } }
			NOT = { has_relation_rival = scope:recipient }
		}
	}

	on_accept = {
		scope:actor = {
			add_character_flag = { flag = jianghu_face_slap_cooldown days = 180 }
			add_prestige = 40
		}

		scope:recipient = {
			add_opinion = {
				target = scope:actor
				modifier = opinion_jianghu_public_humiliation
				years = 80
			}
			add_character_modifier = { modifier = jianghu_face_loss days = 3650 }
			add_stress = 25

			if = {
				limit = {
					is_cultivator = yes
					OR = {
						has_trait = wrathful
						has_trait = vengeful
						has_trait = ambitious
					}
				}
				add_opinion = {
					target = scope:actor
					modifier = opinion_jianghu_face_grudge
					years = 120
				}
			}

			if = {
				limit = { is_ai = no }
				scope:actor = { save_scope_as = xianxia_face_slap_sender }
				trigger_event = { id = xianxia_jianghu_letters.1000 days = 1 }
			}
		}
	}
}
