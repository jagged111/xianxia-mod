	diplomacy = { 5 10 }
	intrigue = { 0 5 }
	stewardship = { 0 10 }
	learning = { 15 25 }
	prowess = { 0 15 } 

	after_creation = {

		add_trait_xp = {
			trait = lifestyle_physician
			value = 300
		}
		add_trait_xp = {
			trait = lifestyle_mystic
			value = { 100 200 }
		}


		# Learn a bunch of languages
		if = {
			limit = {
				NOT = {
					knows_language_of_culture = scope:the_artifact_user.culture
				}
			}
			learn_language_of_culture = scope:the_artifact_user.culture
		}
		if = {
			limit = {
				NOT = {
					knows_language = language_greek
				}
			}
			learn_language = language_greek
		}

		if = {
			limit = {
				NOT = { has_trait = zealous }
			}
			random = {
				chance = 30
				add_secret = { type = secret_non_believer }
			}
		}
		random = {
			chance = 50
			add_secret = { type = secret_witch }
		}
		random = {
			chance = 10
			add_secret = { type = secret_deviant }
		}
		random = {
			chance = 10
			add_trait = adventurer
		}
		random = {
			chance = 5
			add_trait = dwarf
		}

		if = {
			limit = {
				culture = { has_cultural_parameter = has_access_to_shieldmaidens }
			}
			random = {
				chance = 15
				add_trait = shieldmaiden
				add_trait = strong
				hidden_effect = {
					remove_trait = physique_good_1
					remove_trait = physique_good_2
				}
				add_trait = physique_good_3
				add_trait = lifestyle_blademaster
				add_trait_xp = {
					trait = lifestyle_blademaster
					value = 50
				}
			}
		}

	} # end after_creation
}
