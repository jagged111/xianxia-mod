secret_thug_assault = {
	category = thuggery
	
	is_valid = {
		always = yes
	}

	is_shunned = {
		always = no
	}

	is_criminal = {
		always = yes
	}

	on_discover = {
		# Always save scopes immediately for clarity
		scope:secret_target = { save_scope_as = victim }
		scope:secret_owner = { save_scope_as = sender }
	}

	on_expose = {
		scope:secret_target = { save_scope_as = victim }
		scope:secret_owner = { save_scope_as = sender }

		# Trigger event for the owner if they aren't the one exposing it
		if = {
			limit = {
				NOT = { this = scope:secret_owner }
			}
			scope:secret_owner = {
				trigger_event = iti_secrets.1000
			}
		}

		# Apply penalties and traits to the owner (sender)
		scope:sender = {
			add_trait = gallowsbait
			add_trait_xp = { 
				trait = gallowsbait 
				value = 10 
			}
			
			secret_exposed_owner_effects_effect = { 
				SECRET = scope:secret 
				POV = scope:sender 
			}
		}

		secret_exposed_notification_effect = yes

		# Handle House opinion/relation damage
		scope:victim.house = {
			change_house_relation_effect = {
				HOUSE = scope:sender.house
				VALUE = house_relation_damage_medium_value
				REASON = thuggery 
				CHAR = scope:sender
				TARGET_CHAR = scope:victim
			}
		}
	}
}
