convert_to_ks_huxian_culture_decision={
	ai_check_interval = 0
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_culture.dds"
	}
	desc = convert_to_ks_huxian_culture_decision_desc
	selection_tooltip = convert_to_ks_huxian_culture_decision_tooltip
	
	is_shown = {
		always = no
	}
	
	is_valid_showing_failures_only = {
		is_physically_able_adult = yes
		is_at_war = no
		custom_tooltip = {
			text = ks_can_only_change_culture_once_tt
			NOT = { has_character_flag = converted_culture_this_lifetime }
		}
	}
	
	cost = {
		prestige = 500
		piety = 500
		gold = 250
	}
	
	effect={
		if = {
			limit = {
				culture = {
					has_cultural_parameter = less_likely_to_culture_convert
				}
				OR = {
					has_trait = compassionate
					has_trait = gregarious
					has_trait = diligent
					has_trait = fickle
					has_trait = humble
					has_trait = trusting
				}
			}
			add_stress = medium_stress_gain
		}
		else_if = {
			limit = {
				culture = {
					has_cultural_parameter = less_likely_to_culture_convert
				}
			}
			add_stress = major_stress_gain
		}
		
		add_character_flag = {
			flag = converted_culture_this_lifetime
		}
		hidden_effect = {
			add_piety_level = -1
		}
		show_as_tooltip = {
			add_piety_level = -1
		}
		custom_tooltip = convert_to_ks_huxian_culture_decision_tt
	}
	
	#ai_check_interval = 0

	ai_potential = {
		always = no # Only players can do this. AI vassals go through the normal court events/hold court activity.
	}

}
