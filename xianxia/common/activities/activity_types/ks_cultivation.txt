activity_ks_cultivation = {
	name = "activity_ks_cultivation"
	is_shown = {
		#has_bp2_dlc_trigger = yes
		highest_held_title_tier > tier_barony
		is_playable_character = yes
		trigger_if = {
			limit = {
				is_ai = yes
			}
			is_adult = yes
			is_at_war = no
			gold >= 500
			#They should not always do this
			NOT = { has_variable = conqueror }
			trigger_if = { # Throttle Counts
				limit = {
					highest_held_title_tier = tier_county
				}
				ai_energy >= 100
			}
		}
		#Every other activity must be blocked until an uncrowned AI crowns themselves
		trigger_if = {
			limit = {
				is_ai = yes
			}
			NOT = {
				has_realm_law = uncrowned
			}
		}
	}

	can_start_showing_failures_only = {
		NOT = { is_activity_type_on_cooldown = activity_ks_cultivation }
		is_available_adult = yes
		custom_tooltip = {
			text = REASON_NOT_A_REAL_HUXIAN
			OR = {
				has_trait = huxian_blood
				has_trait = huxian_cultivation
			}
		}
	}

	is_valid = {
			scope:host = {
				#Getting imprisoned will invalidate
				is_imprisoned = no
				#Getting unlanded will invalidate
			is_playable_character = yes 
			#Becoming incapable will invalidate
			NOT = { is_incapable = yes }
		}
	}

	on_invalidated = {
		# Host becomes unlanded
		if = {
			limit = {
				scope:host = { is_playable_character = no }
			}
			scope:activity = {
				activity_type = { save_scope_as = activity_type }
			}
			every_attending_character = {
				trigger_event = activity_system.0320
			}
		}
		# Host becomes incapable
		else_if = {
			limit = {
				scope:host = { is_incapable = yes }
			}
			scope:activity = {
				activity_type = { save_scope_as = activity_type }
			}
			scope:host = {
				trigger_event = activity_system.0330
			}
			every_attending_character = {
				limit = { this != scope:host }
				trigger_event = activity_system.0331
			}
		}
		if = {
			limit = {
				scope:host = { is_ai = yes }
			}
			scope:activity = {
				every_attending_character = {
					limit = {
						OR = {
							has_character_flag = ks_teacher_1
							has_character_flag = ks_student_1
							has_character_flag = ks_student_2
						}
					}
					if = {
						limit = {
							is_alive = yes
							has_character_flag = created
							# avoid triggering successions...
							is_ruler = no
						}
						death = {
							death_reason = death_disappearance
						}
					}
				}
			}
		}
	}

	on_host_death = {
		if = {
			limit = {

				every_attending_character = {

							death_reason = death_disappearance
						}
					}
				}
			}

			every_attending_character = {
				limit = {
					has_character_flag = ks_student_1
				}
				remove_character_flag = ks_student_1
			}

				limit = {

		}

			current_travel_plan = {
				every_entourage_character = { set_location_to_default = yes }
			}
		}

					}

					#Set up recurring characters: teachers

					#Set up recurring characters: students

				}
			}

			#Events that happen while we're on site
			on_weekly_pulse = {
				if = {
					limit = { this = scope:host }
					# Let's check that we still have the required teachers and students, and if not let's create them again

					ks_cultivation_teacher_setup = yes
					#Set up recurring characters: students
					ks_cultivation_student_setup = yes
					trigger_event = { on_action = ks_cultivation_destination_events }
				}
			}

			on_end = {
				if = {
					limit = { this = scope:host }
