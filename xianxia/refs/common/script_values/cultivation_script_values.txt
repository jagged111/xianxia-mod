cultivation_progress_multiplier = {

	#Meridian checks
	switch = {
		trigger = has_trait 
		poor_meridians = {
			value = 0.1
		}
		normal_meridians = {
			value = 1
		}
		good_meridians = {
			value = 2.7
		}
		excellent_meridians = {
			value = 4
		}
		heavenly_meridians = {
			value = 50
		}
	}

	#personality checks
	if = {
		limit = {
			has_trait = diligent
		}
		add = 0.1
	} else_if = {
		limit = {
			has_trait = lazy
		}
		add = -0.5
	}

	#constitutions checks
	if = {
		limit = {
			has_trait = yang_constitution
		}
		add = 1
	}
	if = {
		limit = {
			has_trait = yin_constitution
		}
		add = 1
	}
	if = {
		limit = {
			has_trait = poison_constitution
		}
		add = 1
	}
	if = {
		limit = {
			has_trait = lightning_constitution
		}
		add = 1
	}
	if = {
		limit = {
			has_trait = dragon_bloodline
		}
		add = 0.3
	}
	if = {
		limit = {
			has_trait = dragon_body_constitution
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = phoenix_blood_constitution
		}
		add = 1.5
	}
}

chance_to_cultivate = {
	value = 40


	if = {
		limit = {
			is_independent_ruler = no
		}
		add = 30
	}

	if = {
		limit = {
			is_landed = no
		}
		add = 30
	}

	if = {
		limit = {
			is_at_war = yes
		}
		add = -100
	}

	add = cultivation_attitude

	#meditation_exclusive_Attitudes
	if = {

		limit = {
			has_trait = impatient
		}
		add = -15
	}

	#meridians checks
	if = {

		limit = {
			has_trait = poor_meridians
		}
		add = -15
	}
	if = {

		limit = {
			has_trait = good_meridians
		}
		add = 20
	}
	if = {

		limit = {
			has_trait = excellent_meridians
		}
		add = 35
	}
	if = {

		limit = {
			has_trait = heavenly_meridians
		}
		add = 70
	}

}
