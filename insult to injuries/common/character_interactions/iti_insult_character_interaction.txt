iti_insult_interaction = {
	interface_priority = 40
	common_interaction = yes
	category = interaction_category_hostile
	auto_accept = yes

	is_shown = {
		scope:actor = { is_ruler = yes }
		scope:recipient = { 
			is_ruler = yes 
			is_adult = yes
			NOT = { this = scope:actor }
		}
	}

	is_valid = {
		scope:actor = {
			# Check 1: General Insult Cooldown
			custom_description = {
				text = iti_insult_cooldown_tt
				NOT = { has_character_flag = iti_insult_cooldown }
			}
			# Check 2: Active Duel Cooldown 
			custom_description = {
				text = iti_duel_active_cooldown_tt
				NOT = { has_character_flag = iti_duel_cooldown }
			}
		}
	}

	on_accept = {
		
		# CASE 1: AI INSULTS PLAYER/AI
		if = {
			limit = { scope:actor = { is_ai = yes } }
			
			# Save the AI actor for the recipient's event
			scope:actor = { save_scope_as = iti_insult_ruler }
			
			if = {
				limit = { scope:recipient = { is_ai = no } }
				scope:recipient = {
					trigger_event = {
						id = iti_event.1000
						days = { 1 3 }
					}
				}
			}
			else = {
				scope:recipient = {
					trigger_event = {
						id = iti_event.1005
						days = { 1 3 }
					}
				}
			}
		}

		# CASE 2: PLAYER INSULTS AI (Reaction Logic)
		scope:recipient = { 
			add_opinion = {
				target = scope:actor
				modifier = insulted_interaction_opinion
			}

			if = {
				limit = { 
					scope:actor = { is_ai = no }
					is_ai = yes
				}
				
				# Save the AI recipient for the player's event
				save_scope_as = iti_insult_ruler 
				
				add_character_flag = insulted_by_player
				
				random_list = {
					# OUTCOME: COWARD / PASSIVE (Nothing happens)
					30 = {
						modifier = { add = 70 has_trait = craven }
						modifier = { add = 20 has_trait = content }
						modifier = { add = 20 has_trait = humble }
						custom_tooltip = iti_insult_no_retort_tt
					}
					
					# OUTCOME: HONORABLE RETORT (Duel Event 1001)
					35 = {
						modifier = { add = 40 has_trait = brave }
						modifier = { add = 30 has_trait = honest }
						modifier = { add = 20 has_trait = arrogant }
						modifier = { 
							add = 25 
							prowess >= 15 
						}
						modifier = { factor = 0 has_trait = craven } 
						
						custom_tooltip = iti_insult_retort_tt
						scope:actor = {
							trigger_event = {
								id = iti_duel_event.1001
								days = { 30 60 }
							}
						}
					}
					
					# OUTCOME: ESCALATION
					35 = {
						modifier = { add = 40 has_trait = wrathful }
						modifier = { add = 40 has_trait = callous }
						modifier = { add = 20 has_trait = sadistic }
						modifier = { add = 20 has_trait = vengeful }
						
						custom_tooltip = iti_insult_escalation_tt
						scope:actor = {
							trigger_event = {
								id = iti_event.500
								days = { 30 60 }
							}
						}
					}
				}
			}
		}

		# Apply the 3 months cooldown to the sender
		scope:actor = {
			add_character_flag = { flag = iti_insult_cooldown days = 90 }
		}
	}

	
	ai_potential = {
		is_imprisoned = no
		is_at_war = no
	}

	ai_targets = {
		ai_recipients = neighboring_rulers
		ai_recipients = peer_vassals
		max = 5
	}

	ai_frequency_by_tier = {
		barony = 0
		county = 120
		duchy = 60
		kingdom = 30
		empire = 20
		hegemony = 20
	}

	ai_will_do = {
		base = 5
		
		# AI is more likely to insult human players
		modifier = {
			add = 10
			scope:recipient = { is_ai = no }
		}

		# AI is highly likely to insult their rivals
		modifier = {
			add = 30
			has_relation_rival = scope:recipient
		}

		# Jerks insult more often
		modifier = {
			add = 10
			has_trait = wrathful
		}
		modifier = {
			add = 5
			has_trait = arrogant
		}
		modifier = {
			add = 5
			has_trait = sadistic
		}

		# The AI will never randomly insult someone they actually like (unless they are a rival)
		modifier = {
			factor = 0
			opinion = { target = scope:recipient value > 20 }
			NOT = { has_relation_rival = scope:recipient }
		}
	}
}
