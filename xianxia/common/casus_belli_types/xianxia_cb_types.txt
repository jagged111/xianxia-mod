xianxia_seize_spirit_vein_cb = {
    war_name = WAR_NAME
    truce_days = 1825
    can_use = {
        scope:attacker = { is_cultivator = yes }
        scope:defender = { is_landed = yes }
    }
    can_use_title = {
        title_tier = tier_county
    }
    on_success = {
        scope:title = {
            add_county_modifier = { modifier = spirit_vein_secured days = 3650 }
            remove_county_modifier = spirit_vein_contested
        }
        scope:attacker = { change_variable = { name = sect_authority_score add = 6 } }
    }
    on_fail = {
        scope:attacker = { change_variable = { name = sect_authority_score add = -6 } }
    }
}

xianxia_destroy_sect_foundation_cb = {

        }
        scope:defender = {
            add_character_modifier = { modifier = sect_succession_crisis days = 1825 }
            change_variable = { name = sect_authority_score add = -10 }
        }
    }
}
