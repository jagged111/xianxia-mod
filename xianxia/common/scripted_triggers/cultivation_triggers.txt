# Character has a strictly higher cultivation realm than their current liege.
has_higher_cultivation_realm_than_liege = {
	exists = liege
	OR = {
		AND = {
			has_trait = cultivation_true_immortal
			liege = { NOT = { has_trait = cultivation_true_immortal } }
		}
		AND = {
			has_trait = cultivation_entering_nirvana

					has_trait = cultivation_entering_nirvana
