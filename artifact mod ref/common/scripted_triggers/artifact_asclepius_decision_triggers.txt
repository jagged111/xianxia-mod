artifact_asclepius_fix_genes_trigger = {
	scope:the_artifact_user = {
		OR = { # you should have something worth curing
			has_trait = clubfooted
			has_trait = hunchbacked
			has_trait = lisping
			has_trait = stuttering
			has_trait = inbred
			has_trait = spindly
			has_trait = wheezing
			has_trait = bleeder
			NOT = { has_trait = pure_blooded }
		}
	}
}

					has_trait = great_pox
					has_trait = early_great_pox
					has_trait = lovers_pox
					has_trait = leper

			}

					has_trait = early_great_pox
					has_trait = lovers_pox

		}
	}
}
