	icon = de_jure_cb
	group = de_jure

	war_name_base = XIANXIA_CAPTURE_SACRED_MOUNTAIN_WAR_NAME_BASE
	allowed_for_character = { scope:attacker = { is_landed = yes has_cultivation_sect_faith = yes } }
	allowed_against_character = { scope:defender = { is_landed = yes has_cultivation_sect_faith = yes } }
	ai_only_against_neighbors = yes
	should_invalidate = { scope:defender = { is_landed = no } }
	on_invalidated_desc = xianxia_cb_invalidated_desc
	on_victory_desc = xianxia_capture_sacred_mountain_victory_desc
	on_defeat_desc = xianxia_capture_sacred_mountain_defeat_desc
	on_white_peace_desc = xianxia_capture_sacred_mountain_white_peace_desc
	on_victory = {
		scope:attacker.capital_county = { add_county_modifier = { modifier = sacred_mountain_consecrated days = 3650 } add_county_modifier = { modifier = spirit_vein_secured days = 1825 } }
		scope:defender = {
			if = {
				limit = {
					any_sub_realm_county = {
						any_neighboring_county = {
							holder = {
								OR = {
									this = scope:attacker
									any_liege_or_above = { this = scope:attacker }
								}
							}
						}
					}
				}
				random_sub_realm_county = {
					limit = {
						any_neighboring_county = {
							holder = {
								OR = {
									this = scope:attacker
									any_liege_or_above = { this = scope:attacker }
								}
							}
						}
					}
					weight = {
						base = 1
						modifier = {
							add = 250
							any_neighboring_county = { this = scope:attacker.capital_county }
						}
						modifier = {
							add = {
								every_neighboring_county = {
									limit = {
										holder = {
											OR = {
												this = scope:attacker
												any_liege_or_above = { this = scope:attacker }
											}
										}
									}
									add = 35
								}
							}
							any_neighboring_county = {
								holder = {
									OR = {
										this = scope:attacker
										any_liege_or_above = { this = scope:attacker }
									}
								}
							}
						}
					}
					save_scope_as = sacred_mountain_taken_county
				}
			}
			else = {
				random_sub_realm_county = {
					save_scope_as = sacred_mountain_taken_county
				}
			}
		}
		create_title_and_vassal_change = {
			type = conquest
			save_scope_as = change
			add_claim_on_loss = yes
		}
		scope:sacred_mountain_taken_county = {
			change_title_holder = {
				holder = scope:attacker
				change = scope:change
				take_baronies = yes
			}
			add_county_modifier = { modifier = sacred_mountain_consecrated days = 3650 }
			add_county_modifier = { modifier = spirit_vein_secured days = 1825 }
		}
		resolve_title_and_vassal_change = scope:change
		scope:attacker = { add_prestige = 250 add_gold = 320 }
		if = {
			limit = { scope:defender = { highest_held_title_tier = tier_empire } }
			scope:attacker = { add_gold = 320 add_piety = 220 }
			scope:defender = { add_gold = -300 add_prestige = -260 }
		}
		else_if = {
			limit = { scope:defender = { highest_held_title_tier = tier_kingdom } }
			scope:attacker = { add_gold = 220 add_piety = 150 }
			scope:defender = { add_gold = -220 add_prestige = -180 }
		}
		else_if = {
			limit = { scope:defender = { highest_held_title_tier = tier_duchy } }
			scope:attacker = { add_gold = 140 add_piety = 90 }
			scope:defender = { add_gold = -120 add_prestige = -120 }
		}
		else = {
			scope:attacker = { add_gold = 80 add_piety = 45 }
			scope:defender = { add_gold = -70 add_prestige = -70 }
		}
		add_truce_attacker_victory_effect = yes
	}
	on_defeat = { scope:attacker = { add_prestige = -120 } add_truce_attacker_defeat_effect = yes }
	on_white_peace = { add_truce_white_peace_effect = yes }
}
