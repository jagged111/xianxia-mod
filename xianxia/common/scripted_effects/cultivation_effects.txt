cultivation_convert_direct_domain_faith_effect = {
	if = {
		limit = { is_landed = yes }
		# Convert only personally-held county titles (never vassal-held counties).
		every_held_title = {
			limit = { tier = tier_county }
			set_county_faith = root.faith
		}
	}
}

# Shared child-handling flow for fox-spirit awakenings that need parent response logic.
cultivation_fox_spirit_child_parent_resolution_effect = {
	if = {
		limit = { is_adult = no }
		add_character_flag = fox_spirit_parent_decision_pending

		if = {
			limit = { exists = father father = { is_alive = yes is_ai = no } }
			add_character_flag = fox_spirit_waiting_player_parent
			father = { trigger_event = cultivation_ai.2261 }
		}
		if = {
			limit = { exists = mother mother = { is_alive = yes is_ai = no } }
			add_character_flag = fox_spirit_waiting_player_parent
			mother = { trigger_event = cultivation_ai.2261 }
		}

		# AI parents auto-resolve by personality when no player-parent response is pending.
		if = {
			limit = {
				exists = father
				father = { is_alive = yes is_ai = yes }
				has_character_flag = fox_spirit_parent_decision_pending
				NOT = { has_character_flag = fox_spirit_waiting_player_parent }
			}
			if = {
				limit = {
					father = {
						OR = {
							has_trait = just
							has_trait = honest
							has_trait = zealous
						}
					}
				}
				add_character_modifier = { modifier = fox_spirit_disowned_child days = -1 }
				remove_character_flag = fox_spirit_parent_decision_pending
			}
			else = {
				add_opinion = { target = father modifier = opinion_fox_spirit_accepted_child years = 30 }
				remove_character_flag = fox_spirit_parent_decision_pending
			}
		}

		if = {
			limit = {
				exists = mother
				mother = { is_alive = yes is_ai = yes }
				has_character_flag = fox_spirit_parent_decision_pending
				NOT = { has_character_flag = fox_spirit_waiting_player_parent }
			}
			if = {
				limit = {
					mother = {
						OR = {
							has_trait = just
							has_trait = honest
							has_trait = zealous
						}
					}
				}
				add_character_modifier = { modifier = fox_spirit_disowned_child days = -1 }
				remove_character_flag = fox_spirit_parent_decision_pending
			}
			else = {
				add_opinion = { target = mother modifier = opinion_fox_spirit_accepted_child years = 30 }
				remove_character_flag = fox_spirit_parent_decision_pending
			}
		}
	}
}

# Build same-sect defensive pacts among independent cultivator rulers.
cultivation_form_sect_defensive_pacts_effect = {
    if = {
        limit = {
            is_alive = yes
            is_adult = yes
            NOT = { exists = liege }
            is_cultivator = yes
            has_cultivation_sect_faith = yes
        }

        every_independent_ruler = {
            limit = {
                is_alive = yes
                is_adult = yes
                is_cultivator = yes
                has_cultivation_sect_faith = yes
                faith = root.faith
                NOT = { this = root }
                NOT = { is_allied_to = root }
                OR = {
                    is_within_diplo_range = root
                    AND = {
                        root = {
                            NOT = { has_trait = cultivation_qi_gathering }
                        }
                        NOT = { has_trait = cultivation_qi_gathering }
                    }
                }
            }
            add_alliance = root
        }
    }
}

        limit = { faith = { has_doctrine_parameter = orthodox_cultivation } }
        add_character_modifier = { modifier = sect_signature_orthodox_hidden days = -1 }
    }
    else_if = {
        limit = { faith = { has_doctrine_parameter = demonic_cultivation } }
        add_character_modifier = { modifier = sect_signature_demonic_hidden days = -1 }
    }
    else_if = {
        limit = { faith = { has_doctrine_parameter = unorthodox_cultivation } }
        add_character_modifier = { modifier = sect_signature_unorthodox_hidden days = -1 }
    }
    else_if = {
        limit = { faith = { has_doctrine_parameter = vagrant_cultivation } }
        add_character_modifier = { modifier = sect_signature_vagrant_hidden days = -1 }
    }
}
