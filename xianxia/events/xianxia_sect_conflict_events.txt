namespace = xianxia_sect_conflict

# Monthly sect-politics maintenance and pressure pulse.
xianxia_sect_conflict.0001 = {
    type = character_event
    scope = none
    hidden = yes

    immediate = {
        every_ruler = {
            limit = {
                is_alive = yes
                is_adult = yes
                is_landed = yes
                is_cultivator = yes
                has_cultivation_sect_faith = yes
            }

            xianxia_update_sect_authority_effect = yes
            xianxia_apply_lineage_backing_effect = yes

            if = {
                limit = {
                    var:sect_authority_score < 38
                    any_courtier = {
                        is_alive = yes
                        is_adult = yes
                        is_cultivator = yes
                        faith = root.faith
                    }
                    NOT = { has_character_flag = sect_internal_conflict_recent }
                }
                random = {
                    chance = 16
                    trigger_event = xianxia_sect_conflict.0100
                }
            }

            if = {
                limit = { var:sect_authority_score >= 62 }
                capital_county = {
                    remove_county_modifier = spirit_vein_contested
                    if = {
                        limit = { NOT = { has_county_modifier = spirit_vein_secured } }
                        add_county_modifier = { modifier = spirit_vein_secured days = 1095 }
                    }
                }
            }
            else_if = {
                limit = { var:sect_authority_score <= 34 }
                capital_county = {
                    remove_county_modifier = spirit_vein_secured
                    add_county_modifier = { modifier = spirit_vein_contested days = 1095 }
                    change_control = -10
                }
            }
        }
    }

    option = { name = OK }
}

# Elder faction unrest event.
xianxia_sect_conflict.0100 = {
    type = character_event
    title = xianxia_sect_conflict.0100.t
    desc = xianxia_sect_conflict.0100.desc

    option = {
        name = xianxia_sect_conflict.0100.a
        add_gold = -80
        change_variable = { name = sect_authority_score add = 8 }
        add_character_modifier = { modifier = sect_authority_high days = 1460 }
        add_character_flag = { flag = sect_internal_conflict_recent days = 1095 }
        every_courtier = {
            limit = {
                is_alive = yes
                is_adult = yes
                is_cultivator = yes
                faith = root.faith
            }
            add_opinion = { target = root modifier = opinion_sect_leader_appeased_elders years = 10 }
        }
    }

    option = {
        name = xianxia_sect_conflict.0100.b
        change_variable = { name = sect_authority_score add = 4 }
        add_character_modifier = { modifier = sect_purge_aftermath days = 1825 }
        add_character_flag = { flag = sect_internal_conflict_recent days = 1460 }
        random_courtier = {
            limit = {
                is_alive = yes
                is_adult = yes
                is_cultivator = yes
                faith = root.faith
                NOT = { has_trait = sect_leader }
            }
            add_character_modifier = { modifier = sect_exiled_elder days = 3650 }
            add_stress = 80
            add_opinion = { target = root modifier = opinion_sect_leader_purged_elders years = 20 }
        }
        every_courtier = {
            limit = {
                is_alive = yes
                is_adult = yes
                is_cultivator = yes
                faith = root.faith
            }
            random_list = {
                55 = { add_opinion = { target = root modifier = opinion_sect_leader_purged_elders years = 15 } }
                45 = { add_opinion = { target = root modifier = opinion_sect_leader_crushed_dissent years = 15 } }
            }
        }
    }

    option = {
        name = xianxia_sect_conflict.0100.c
        trigger_event = xianxia_sect_conflict.0110
        add_character_flag = { flag = sect_internal_conflict_recent days = 1825 }
    }
}

# Challenge resolution.
xianxia_sect_conflict.0210 = {
    type = character_event
    title = xianxia_sect_conflict.0210.t
    desc = xianxia_sect_conflict.0210.desc

    immediate = {
        if = {
            limit = {
                exists = scope:sect_leader_target
                OR = {
                    prowess >= scope:sect_leader_target.prowess
                    martial >= scope:sect_leader_target.martial
                }
            }
            random_list = {
                68 = { trigger_event = xianxia_sect_conflict.0220 }
                32 = { trigger_event = xianxia_sect_conflict.0230 }
            }
        }
        else = {
            random_list = {
                32 = { trigger_event = xianxia_sect_conflict.0220 }
                68 = { trigger_event = xianxia_sect_conflict.0230 }
            }
        }
    }

    option = {
        name = xianxia_sect_conflict.0210.a
    }
}

# Challenger victory: forced sect leadership change.
xianxia_sect_conflict.0220 = {
    type = character_event
    title = xianxia_sect_conflict.0220.t
    desc = xianxia_sect_conflict.0220.desc

    option = {
        name = xianxia_sect_conflict.0220.a

        add_character_modifier = { modifier = sect_challenger_glory days = 3650 }
        add_character_flag = { flag = sect_challenge_cooldown days = 3650 }
        add_prestige = 400
        add_dynasty_prestige = 150

        remove_trait = sect_disciple
        remove_trait = sect_outer_hall_elder
        remove_trait = sect_enforcement_elder
        add_trait = sect_leader

        if = {
            limit = { exists = scope:sect_leader_target }
            scope:sect_leader_target = {
                remove_trait = sect_leader
                add_trait = sect_outer_hall_elder
                add_character_modifier = { modifier = sect_forced_abdication_shame days = 3650 }
                add_stress = 90
                add_opinion = { target = root modifier = opinion_sect_challenger_usurped_leadership years = 30 }
            }
        }

        every_courtier = {
            limit = {
                is_alive = yes
                is_adult = yes
                is_cultivator = yes
                faith = root.faith
            }
            random_list = {
                50 = { add_opinion = { target = root modifier = opinion_sect_challenger_usurped_leadership years = 20 } }
                50 = { add_opinion = { target = root modifier = opinion_sect_leader_crushed_dissent years = 20 } }
            }
        }
    }
}
