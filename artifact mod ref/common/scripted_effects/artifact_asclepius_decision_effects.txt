artifact_asclepius_create_effect = {
	$OWNER$ = { save_scope_as = owner }
	
	hidden_effect = {
		random_county_in_region = {
			region = world_europe_south_east
			random_county_province = {
				save_scope_as = artifact_found_here
			}
		}
	} # end hidden_effect
	
	hidden_effect_new_object = {
		scope:owner = {
			save_scope_value_as = {
				name = wealth
				value = 500
			}
			save_scope_value_as = {
				name = quality
				value = 500
			}
			create_artifact = {
				name = artifact_asclepius_name
				description = artifact_asclepius_desc

				rarity = illustrious
				
				type = regalia
				
				# big health, fertility, etc boosts
				
				visuals = regalia
		
				template = artifact_asclepius_template
		
				modifier = artifact_monthly_prestige_8_modifier
				modifier = artifact_asclepius_modifier
		
				quality = scope:quality
				wealth = scope:wealth
		
				history = {
					location = scope:artifact_found_here
					actor = scope:owner
					recipient = scope:owner
					type = discovered
				}

				decaying = no
				save_scope_as = newly_created_artifact
			} # end create_artifact
		} # end scope:owner
		scope:newly_created_artifact = {
			set_variable = {
				name = quality
				value = scope:quality
			}
			set_variable = {
				name = wealth
				value = scope:wealth
			}
			get_artifact_feature_references_effect = yes
		}

	} # end hidden_effect_new_object
} # end artifact_asclepius_create_effect


# Recruit Doctor
artifact_asclepius_recruit_doctor_effect = {

	custom_tooltip = artifact_asclepius_recruit_doctor_effect_tooltip

	hidden_effect = {
		# Grab a culture for the doctor
		#	This should favor close cultures, cultures with compassion, and cultures
		#	with herbalists
		scope:the_artifact_user = {
			every_sub_realm_county = {
				this.culture = {
					add_to_list = nearby_cultures	
				}
				
			}
			every_character_to_title_neighboring_and_across_water_county = {
				this.culture = {
					add_to_list = nearby_cultures
				}
			}
		} # end scope:the_artifact_user
		every_culture_global = {
			limit = {
				OR = {
					has_cultural_parameter = herbalist_traits_more_common
					has_cultural_parameter = compassionate_trait_more_common
				}
			}
			add_to_list = nearby_cultures
		}

		random_in_list = {
			list = nearby_cultures
			save_scope_as = the_doctor_culture
		}
		# clear the list
		clear_variable_list = nearby_cultures
		
		create_character = {
			#location = scope:the_artifact_user.capital_province
			template = artifact_asclepius_doctor
			save_scope_as = the_doctor
			employer = scope:the_artifact_user
			dynasty = none
		}

		scope:the_doctor = {
			#make my friend
			set_relation_friend = scope:the_artifact_user
		}		
	} #end hidden_effect

	# send a toast
	scope:the_artifact_user = {
		send_interface_toast = {
			type = character_arrived
			title = msg_artifact_asclepius_recruited_doctor_title
			desc = msg_artifact_asclepius_recruited_doctor_desc
			left_icon = scope:the_doctor
		}
	}
}

# Adds one level of good physique
artifact_asclepius_pump_me_up_effect = {
	scope:the_artifact_user = {
		if = {
			limit = { has_trait = weak }
			remove_trait = weak
		}
		else_if = {
			limit = { has_trait = physique_bad_1 }
			remove_trait = physique_bad_1
		}
		else_if = {
			limit = { has_trait = physique_bad_2 }
			add_trait_force_tooltip = physique_bad_1
			remove_trait = physique_bad_2
			add_trait = physique_bad_1
		}
		else_if = {
			limit = { has_trait = physique_bad_3 }
			add_trait_force_tooltip = physique_bad_2
			remove_trait = physique_bad_3
			add_trait = physique_bad_2
		}		
		else_if = {
			limit = { has_trait = physique_good_1 }
			add_trait_force_tooltip = physique_good_2
			hidden_effect = { remove_trait = physique_good_1 }
			add_trait = physique_good_2
		}
		else_if = {
			limit = { has_trait = physique_good_2 }
			add_trait_force_tooltip =  physique_good_3
			hidden_effect = { remove_trait = physique_good_2 }
			add_trait = physique_good_3
		}
		else_if = {
			# If you have both physique_good_3 and strong, 
			#  the event should not show up at all.
			limit = { 
				has_trait = physique_good_3
				NOT = { has_trait = strong } 
			}
			add_trait = strong
		}
		else = {
			add_trait = physique_good_1
		}
	}
}
