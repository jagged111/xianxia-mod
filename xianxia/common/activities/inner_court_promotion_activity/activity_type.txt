activity_inner_court_promotion = {
	is_shown = {
		is_adult = yes
		is_landed = yes
		is_cultivator = yes
	}

	can_start_showing_failures_only = {
		NOT = { is_activity_type_on_cooldown = activity_inner_court_promotion }
		is_available_adult = yes
		is_landed = yes
		is_cultivator = yes
	}

	is_valid = {
		scope:host = {
			is_alive = yes
			is_capable_adult = yes
			is_imprisoned = no
			is_landed = yes
			is_cultivator = yes
			any_courtier = {
				is_alive = yes
				is_adult = yes
				has_trait = sect_disciple
				OR = {
					has_trait = good_meridians
					has_trait = excellent_meridians
					has_trait = heavenly_meridians
				}
			}
		}
	}

	on_invalidated = {
		invalidation_imprisoned_effect = yes
		invalidation_incapable_effect = yes
	}

	province_filter = realm
	ai_province_filter = realm
	max_province_icons = 1
	allow_zero_guest_invites = yes
	open_invite = no
	cooldown = { years = 8 }

	can_be_activity_guest = {
		is_alive = yes
		is_adult = yes
		is_cultivator = yes
	}

	phases = {
		inner_court_promotion_phase_review = {
			is_predefined = yes
			location_source = pickable
			on_phase_active = {
				if = {
					limit = { this = scope:host }
					trigger_event = xianxia_activity.7400
					scope:activity = { progress_activity_phase_after = { days = 10 } }
				}
			}
		}

		inner_court_promotion_phase_trials = {
			is_predefined = yes
			on_phase_active = {
				if = {
					limit = { this = scope:host }
					trigger_event = xianxia_activity.7410
					scope:activity = { progress_activity_phase_after = { days = 8 } }
				}
			}
		}

		inner_court_promotion_phase_investiture = {
			is_predefined = yes
			on_phase_active = {
				if = {
					limit = { this = scope:host }
					trigger_event = xianxia_activity.7420
				}
			}
		}
	}
}
