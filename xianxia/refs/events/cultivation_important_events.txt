cultivation_important_event.001 = { # Get meridians
	type = character_event
	title = cultivation_important_event.001.title
	desc = cultivation_important_event.001.desc

	theme = faith
	#content_source = xianxia_cultivation

	left_portrait = {
		character = ROOT
		animation = thinking
	}

	option = { # Confirm
        name = cultivation_important_event.001.a
        ai_chance = {
			base = 2
		}
        random_list = {
            100 = {}
            100 = {add_trait = poor_meridians}
            70 = {add_trait = normal_meridians}
            40 = {add_trait = good_meridians}
            8 = {add_trait = excellent_meridians}
            1 = {add_trait = heavenly_meridians}
        }
	}
    option = { #pay for better odds
        name = cultivation_important_event.001.b

        remove_short_term_gold = 5000
        ai_chance = {
			base = 0.1
		}

        random_list = {
            80 = {}
            30 = {add_trait = poor_meridians}
            70 = {add_trait = normal_meridians}
            100 = {add_trait = good_meridians}
            15 = {add_trait = excellent_meridians}
            1 = {add_trait = heavenly_meridians}
        }
    }
    option = { # nvm
        name = cultivation_important_event.001.c
        ai_chance = {
			base = 1
		}
    }
}

#Breakthrough Events

cultivation_important_event.011 = {
    type = character_event
    title = cultivation_important_event.011.title
    desc = cultivation_important_event.011.desc
    trigger = {
        is_cultivator = yes
    }
    theme = faith
    right_portrait = {
        character = ROOT
        animation = happiness
    }

    immediate = {
        play_music_cue = "mx_cue_positive_effect"
		random_artifact = {
			limit = {
				artifact_owner = root
				has_variable = is_breakthrough_herb
			}
			save_scope_as = breakthrough_herb
		}
	}

    option = {
        name = cultivation_important_event.011.a
        ai_chance = {
			base = 2
		}
        random_list = {
            15 = {
                add_trait_xp = {
                    trait = qi_gathering
                    value = -20
                }
            }
            83 = {
                breakthrough_effect = yes
                modifier = {
					add = breakthrough_difficulty_modifier
				}
            }
            20 = {
                death = {
                    death_reason = death_breakthrough
                }
            }
        }
    }
    option = { # Use a herb for better odds
        name = cultivation_important_event.011.b
        destroy_artifact = scope:breakthrough_herb
        ai_chance = {
			base = 10
		}
        trigger = {
            scope:breakthrough_herb = {
                root = artifact_owner #TODO: not a great check, figure out how to improve this later
            }
        }
        random_list = {
            5 = {
                add_trait_xp = {
                    trait = qi_gathering
                    value = -20
                }
            }
            94 = {
                breakthrough_effect = yes
                modifier = {
					add = breakthrough_difficulty_modifier
				}
            }
            20 = {
                death = {
                    death_reason = death_breakthrough
                }
            }
        }
    }
    option = {
        name = cultivation_important_event.011.c
    }
}
