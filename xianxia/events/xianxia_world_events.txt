xianxia_world.1000 = {
    type = character_event
    scope = none
    hidden = yes

    immediate = {
        every_ruler = {
            limit = {
                is_alive = yes
                is_landed = yes
                is_adult = yes
            }
            xianxia_apply_government_adaptation_effect = yes
        }

        every_ruler = {
            limit = {
                is_alive = yes
                is_adult = yes
                is_cultivator = yes
            }

            if = {
                limit = {
                    OR = {
                        has_trait = cultivation_nascent_soul
                        has_trait = cultivation_heavenly_being
                        has_trait = cultivation_four_axis
                        has_trait = cultivation_integration
                        has_trait = cultivation_star_shattering
                        has_trait = cultivation_sacred_vessel
                        has_trait = cultivation_entering_nirvana
                        has_trait = cultivation_true_immortal
                    }
                }
                add_character_modifier = {
                    modifier = high_realm_longevity
                    days = 365
                }
            }
        }
    }

    option = { name = OK }
}

# Sect institutions and internal governance maintenance.
xianxia_world.1700 = {
    type = character_event
    scope = none
    hidden = yes

    immediate = {
        every_ruler = {
            limit = {
                is_alive = yes
                is_adult = yes
                is_landed = yes
                is_cultivator = yes
                has_cultivation_sect_faith = yes
            }

            if = {
                limit = { NOT = { has_trait = sect_leader } }
                add_trait = sect_leader
            }

            every_courtier = {
                limit = {
                    is_alive = yes
                    is_adult = yes
                    is_cultivator = yes
                    faith = prev.faith
                    NOT = { has_trait = sect_leader }
                }
                if = {
                    limit = {
                        OR = {
                            has_trait = cultivation_core_formation
                            has_trait = cultivation_nascent_soul
                            has_trait = cultivation_heavenly_being
                            has_trait = cultivation_four_axis
                            has_trait = cultivation_integration
                            has_trait = cultivation_star_shattering
                            has_trait = cultivation_sacred_vessel
                            has_trait = cultivation_entering_nirvana
                            has_trait = cultivation_true_immortal
                        }
                    }
                    random_list = {
                        35 = {
                            remove_trait = sect_outer_hall_elder
                            remove_trait = sect_enforcement_elder
                            remove_trait = sect_disciple
                            add_trait = sect_inner_hall_elder
                        }
                        40 = {
                            remove_trait = sect_inner_hall_elder
                            remove_trait = sect_enforcement_elder
                            remove_trait = sect_disciple
                            add_trait = sect_outer_hall_elder
                        }
                        25 = {
                            remove_trait = sect_inner_hall_elder
                            remove_trait = sect_outer_hall_elder
                            remove_trait = sect_disciple
                            add_trait = sect_enforcement_elder
                        }
                    }
                }
                else = {
                    remove_trait = sect_inner_hall_elder
                    remove_trait = sect_outer_hall_elder
                    remove_trait = sect_enforcement_elder
                    add_trait = sect_disciple
                }
            }

            random = {
                chance = 28
                add_character_modifier = { modifier = sect_resource_stockpile days = 365 }
            }
            random = {
                chance = 20
                add_character_modifier = { modifier = sect_resource_obligations days = 365 }
            }
        }
    }
    option = { name = OK }
}
