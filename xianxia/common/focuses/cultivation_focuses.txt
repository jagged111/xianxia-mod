cultivation_orthodox_focus = {
	lifestyle = cultivation_lifestyle
	
	is_valid = {
		# DOCTRINE RESTRICTION: Must have orthodox cultivation doctrine
		faith = {
			has_doctrine_parameter = orthodox_cultivation
		}
	}
	
	desc = {
		desc = cultivation_orthodox_focus_desc
		desc = line_break
	}
	
	modifier = {
		monthly_prestige = 0.5
		same_faith_opinion = 10
		learning = 1
		stress_loss_mult = 0.1
	}
	
	auto_selection_weight = {
		value = 11
		if = {
			limit = {
				OR = {
					has_trait = just
					has_trait = humble
					has_trait = temperate
				}
			}
			add = 200
		}
		if = {
			limit = {
				OR = {
					has_trait = wrathful
					has_trait = sadistic
					has_trait = callous
				}
			}
			add = -200
		}
	}
}
