## The Jolly Roger Artifact!


artifact_jolly_roger_create_effect = {
	# Sets up the variable that changes the description for the jolly roger event,
	#	so it can cycle between several different songs.
	set_global_variable = {
		name = jolly_roger_use_counter
		value = 0
	}

	create_artifact = {	
		name = artifact_jolly_roger_name
		description = artifact_dolly_roger_desc
		rarity = illustrious
		type = wall_big

		#visuals = banner
		visuals = jolly_banner_07 # This makes it always the triangular banner
		visuals_source = title:k_pirates
		
		modifier = artifact_court_grandeur_baseline_add_6_modifier
		modifier = artifact_jollyroger_modifier
		
		save_scope_as = newly_created_artifact
		decaying = no

		quality = 1000
		wealth = 1000
	}
} # end artifact_jolly_roger_create_effect

artifact_jolly_roger_build_ports_effect = {

	#desc = "Gives free ports."
	#custom_tooltip = "Ports."

	#builds a free port in the area
	scope:the_waver = {
		every_sub_realm_county = {
			every_county_province = {
				limit = {
					building_common_tradeport_requirement_terrain = yes
					building_requirement_castle_city_church = { LEVEL = 01 }				## does not have tradeport
					NOT = { has_building_or_higher = common_tradeport_01 }
				}
				add_province_modifier = extra_building_slot

				add_building = common_tradeport_02
			}
		}
	}
} # end artifact_jolly_roger_build_ports_effect

jolly_roger_recruit_champions_effect = {
	custom_tooltip = jolly_roger_recruit_champions_effect_tooltip

	hidden_effect = {
		#Compute culture
		random_list = {
			30 = {
				scope:the_waver.culture = {
					save_scope_as = the_raider_culture
				}
			}
			10 = {
				scope:the_waver = { # Sometimes a culture next door.
					random_character_to_title_neighboring_and_across_water_county = {
						this.culture = {
							save_scope_as = the_raider_culture
						}
					}
				}
			}
			20 = {
				random_culture_global = {
					limit = { # Rowdy cultures should be more likely to generate raiders.
						OR = {
							has_cultural_parameter = rowdy_trait_more_common
							has_cultural_parameter = has_access_to_shieldmaidens
						}
					}
					save_scope_as = the_raider_culture
				}				
			}				
		} # end random_list

		create_character = {
			#location = scope:the_waver.capital_province
			template = artifact_jolly_roger_champion
			dynasty = none
			employer = scope:the_waver
			save_scope_as = the_raider
		}

		scope:the_raider = {
			set_relation_friend = scope:the_waver
		}
	}

	scope:the_waver = {
		send_interface_toast = {
			type = character_arrived
			title = msg_artifact_jolly_roger_raider_title
			desc = msg_artifact_jolly_roger_raider_desc
			left_icon = scope:the_raider
		}		
	}
} # end jolly_roger_recruit_champions_effect

artifact_jolly_roger_stolen_city_effect = {
	if = {
		limit = {
			NOT = {
				scope:the_waver.capital_county = {
					has_county_modifier = fp1_mann_pirate_capital_modifier
				}
			}
		}
		scope:the_waver.capital_county = {
			add_county_modifier = {
				modifier = fp1_mann_pirate_capital_modifier
			}
		}
	} # fi
	else = {
		random_sub_realm_county = {
			limit = { 
				holder = scope:the_waver #only directly heald counties
				NOT = { has_county_modifier = fp1_mann_pirate_capital_modifier }
			} 
			add_county_modifier = {
				modifier = fp1_mann_pirate_capital_modifier
			}
		}
	} # end else
} # artifact_jolly_roger_stolen_city_effect
