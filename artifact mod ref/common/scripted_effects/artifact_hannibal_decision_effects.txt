	create_artifact = {

		visuals = human_skull

# add general
artifact_hannibal_recruit_general_effect = {

	custom_tooltip = artifact_hannibal_recruit_general_effect_tooltip

	#Compute culture
	hidden_effect = {
		random_list = {
			10 = {
				scope:the_artifact_user.culture = {
					save_scope_as = the_recruit_culture
				}
			}
			10 = {
				random_character_to_title_neighboring_and_across_water_county = {
					this.culture = {
						save_scope_as = the_recruit_culture
					}
				}
			}
			20 = {
				random_culture_global = {
					limit = { # Traits with better commanders
						has_cultural_parameter = commanders_likelier_to_gain_commander_traits
					}
					save_scope_as = the_recruit_culture
				}				
			}				
			10 = {
				random_culture_global = {
					limit = {
						has_cultural_parameter = has_access_to_shieldmaidens
					}
					save_scope_as = the_recruit_culture
				}				
			}
			
		} #end random_list	
		
		create_character = {
			#location = scope:the_artifact_user.capital_province
			template = artifact_hannibal_general
			save_scope_as = the_general
			employer = scope:the_artifact_user
			dynasty = none
		}

		scope:the_general = {
			set_relation_friend = scope:the_artifact_user
		}
	} #end hidden_effect

	# send a toast
	send_interface_toast = {
		type = character_arrived
		title = msg_artifact_hannibal_general_title
		desc = msg_artifact_hannibal_general_desc
		left_icon = scope:the_general
	}
}

# add barracks and camps
artifact_hannibal_build_camps_effect = {
	scope:the_artifact_user = {
		every_sub_realm_county = {
			every_county_province = {
				# Tribals first
				if = {
					limit = {
						has_holding_type = tribal_holding				
					}
					if = {
						limit = {
							NOT = { has_building_or_higher = longhouses_01 }
							NOT = { free_building_slots = 0 }
						}
						add_building = longhouses_02
					}
					if = {
						limit = {
							NOT = { has_building_or_higher = war_camps_01 }
							NOT = { free_building_slots = 0 }
						}
						add_building = war_camps_02
					}
				}

				# Military Camps
				if = {
					limit = {
						NOT = { has_building_or_higher = military_camps_01 }
						building_military_camps_requirement_terrain = yes
						building_requirement_castle_city_church = { LEVEL = 01 }
					}
					if = {
						limit = {
							OR = {
								free_building_slots = 0
								#building_slots < 6
							}
						}
						add_province_modifier = extra_building_slot

					}
					add_building = military_camps_02
				}

				if = {
					limit = {
						NOT = { has_building_or_higher = regimental_grounds_01 }
						building_regimental_grounds_requirement_terrain = yes
						building_requirement_castle_city_church = { LEVEL = 01 }
					}
					if = {
						limit = {
							OR = {
								free_building_slots = 0
								#building_slots < 6
							}
						}
						add_province_modifier = extra_building_slot

					}
					add_building = regimental_grounds_02
				}

				if = {
					limit = {
						NOT = { has_building_or_higher = barracks_01 }
						building_barracks_requirement_terrain = yes
						building_requirement_castle_city_church = { LEVEL = 01 }
					}
					if = {
						limit = {
							OR = {
								free_building_slots = 0
								#building_slots < 6
							}
						}
						add_province_modifier = extra_building_slot

					}
					add_building = barracks_02
				}

				if = {
					limit = {
						NOT = { has_building_or_higher = camel_farms_01 }
						building_camel_farms_requirement_terrain = yes
						building_requirement_castle_city_church = { LEVEL = 01 }
					}
					if = {
						limit = {
							OR = {
								free_building_slots = 0
								#building_slots < 6
							}
						}
						add_province_modifier = extra_building_slot

					}
					add_building = camel_farms_02
				}

				if = {
					limit = {
						NOT = { has_building_or_higher = elephant_pens_01 }
						building_elephant_pens_requirement_terrain = yes
