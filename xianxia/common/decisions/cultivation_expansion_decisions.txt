    picture = { reference = "gfx/interface/illustrations/decisions/decision_misc.dds" }

    confirm_text = awaken_meridian_disciples_decision_confirm
    selection_tooltip = awaken_meridian_disciples_tooltip

    is_shown = {
        is_cultivator = yes
        OR = { is_ruler = yes has_court_position = court_tutor }
        OR = {
            any_courtier = { is_adult = yes NOT = { is_cultivator = yes } OR = { has_trait = poor_meridians has_trait = normal_meridians has_trait = good_meridians has_trait = excellent_meridians has_trait = heavenly_meridians } }
            any_vassal = { is_adult = yes NOT = { is_cultivator = yes } OR = { has_trait = poor_meridians has_trait = normal_meridians has_trait = good_meridians has_trait = excellent_meridians has_trait = heavenly_meridians } }
        }
        NOT = { has_character_flag = awaken_meridian_disciples_cooldown }
    }

    is_valid = { always = yes }
    ai_check_interval = 60
    ai_potential = { always = yes }
    ai_will_do = { base = 50 }

    effect = {
        every_courtier = {
            limit = {
                is_adult = yes
                NOT = { is_cultivator = yes }
                OR = { has_trait = poor_meridians has_trait = normal_meridians has_trait = good_meridians has_trait = excellent_meridians has_trait = heavenly_meridians }
            }
            add_trait = cultivation_qi_gathering
            set_character_faith = root.faith
        }
        every_vassal = {
            limit = {
                is_adult = yes
                NOT = { is_cultivator = yes }
                OR = { has_trait = poor_meridians has_trait = normal_meridians has_trait = good_meridians has_trait = excellent_meridians has_trait = heavenly_meridians }
            }
            add_trait = cultivation_qi_gathering
            set_character_faith = root.faith
        }

        add_piety = 150
        add_prestige = 200
        add_character_flag = { flag = awaken_meridian_disciples_cooldown days = 1825 }
    }
}
