activity_jianghu_grand_gathering = {
	is_shown = {
		is_adult = yes
		is_landed = yes
		is_cultivator = yes
	}

	can_start_showing_failures_only = {
		NOT = { is_activity_type_on_cooldown = activity_jianghu_grand_gathering }
		is_available_adult = yes
		is_landed = yes
		is_cultivator = yes
	}

	is_valid = {
		scope:host = {
			is_alive = yes
			is_capable_adult = yes
			is_imprisoned = no
			is_landed = yes
			is_cultivator = yes
		}
	}

	on_invalidated = {
		invalidation_imprisoned_effect = yes
		invalidation_incapable_effect = yes
	}

	province_filter = realm
	ai_province_filter = realm
	max_province_icons = 1
	allow_zero_guest_invites = no
	open_invite = no

	cooldown = { years = 5 }

	can_be_activity_guest = {
		is_alive = yes
		is_adult = yes
		is_cultivator = yes
	}

	phases = {
		jianghu_gathering_phase_arrival = {
			is_predefined = yes
			location_source = pickable
			on_phase_active = {
				if = {
					limit = { this = scope:host }
					scope:activity = { progress_activity_phase_after = { days = 10 } }
					add_prestige = 75
				}
			}
		}

		jianghu_gathering_phase_sparring = {
			is_predefined = yes
			on_phase_active = {
				if = {
					limit = { this = scope:host }
					scope:activity = { progress_activity_phase_after = { days = 12 } }
					if = {
						limit = { prowess >= 22 }
						add_prestige = 120
						add_piety = 80
					}
					else_if = {
						limit = { prowess >= 14 }
						add_prestige = 70
					}
					else = {
						add_stress = 30
						add_prestige = -40
					}
				}
			}
		}

		jianghu_gathering_phase_exchange = {
			is_predefined = yes
			on_phase_active = {
				if = {
					limit = { this = scope:host }
					scope:activity = { progress_activity_phase_after = { days = 10 } }
					add_character_modifier = { modifier = jianghu_activity_host_momentum days = 730 }
				}
			}
		}

		jianghu_gathering_phase_climax = {
			is_predefined = yes
			on_phase_active = {
				if = {
					limit = { this = scope:host }
					add_prestige = 220
					add_character_modifier = { modifier = jianghu_activity_host_momentum days = 1095 }
					change_variable = { name = jianghu_tourney_wins_total add = 1 }

					if = {
						limit = { faith = { has_doctrine_parameter = orthodox_cultivation } }
						change_variable = { name = jianghu_tourney_wins_orthodox add = 1 }
					}
					else_if = {
						limit = { faith = { has_doctrine_parameter = unorthodox_cultivation } }
						change_variable = { name = jianghu_tourney_wins_unorthodox add = 1 }
					}
					else_if = {
						limit = { faith = { has_doctrine_parameter = demonic_cultivation } }
						change_variable = { name = jianghu_tourney_wins_demonic add = 1 }
					}
					else_if = {
						limit = { faith = { has_doctrine_parameter = vagrant_cultivation } }
						change_variable = { name = jianghu_tourney_wins_vagrant add = 1 }
					}
				}
			}
		}
	}
}
