# Mount Hua Sect - Join Event 1000
cultivation_sect_join.1000 = {
    type = character_event
    title = cultivation_sect_join.1000.t
    desc = cultivation_sect_join.1000.desc
    theme = martial

    left_portrait = { character = ROOT animation = ecstasy }
    right_portrait = { character = scope:sect_elder animation = personality_bold }

    # MASTER OPTION - High cultivation level required
    option = {
        name = sect_join
		custom_tooltip = sect_join_is_cultivator_desc
		
        trigger = { 
			custom_tooltip = {
				text = sect_join_isnt_cultivator_desc
				is_cultivator= yes 
			}
		}
			
        show_as_unavailable = { always = yes }
			
        set_character_faith = faith:mount_hua_sect
        cultivation_convert_direct_domain_faith_effect = yes
        add_character_flag = sect_join_success
    }

    # DONATION OPTION - Pay gold to join
    option = {
        name = sect_join_donate
        trigger = { short_term_gold >= 200 }
        show_as_unavailable = { short_term_gold < 200 }
        remove_short_term_gold = 200
        set_character_faith = faith:mount_hua_sect
        cultivation_convert_direct_domain_faith_effect = yes
        add_character_flag = sect_join_success
    }

    # CANCEL
    option = {
        name = sect_join_cancel
    }
}
