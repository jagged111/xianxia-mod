namespace = xianxia_activity

xianxia_activity.7300 = {
	type = character_event
	title = xianxia_activity.7300.t
	desc = xianxia_activity.7300.desc
	theme = feast_activity

	option = {
		name = xianxia_activity.7300.a
		add_prestige = 60

		# First attempt: recruit talented disciples from within the ruler's realm/court.
			random_courtier = {
				limit = {
					is_alive = yes
					is_adult = yes
					faith = root.faith
					NOT = { is_ruler = yes }
					NOT = { has_trait = sect_elder }
				OR = {
					has_trait = good_meridians
					has_trait = excellent_meridians
					has_trait = heavenly_meridians
				}
			}
			save_scope_as = sect_exam_inner_recruit
		}

		scope:sect_exam_inner_recruit ?= {
			if = { limit = { NOT = { has_trait = sect_disciple } } add_trait = sect_disciple }
			remove_character_modifier = outer_court_disciple
			add_character_modifier = { modifier = inner_court_disciple days = 5475 }
			if = {
				limit = {
					NOT = {
						OR = {
							has_trait = weapon_combat_talent_1
							has_trait = weapon_combat_talent_2
							has_trait = weapon_combat_talent_3
							has_trait = weapon_combat_talent_4
							has_trait = weapon_combat_talent_5
						}
					}
				}
				xianxia_apply_weapon_talent_lottery_effect = yes
			}
		}

		# If no fitting inner-court recruit exists in realm/court, recruit a talented wanderer.
		if = {
			limit = { NOT = { exists = scope:sect_exam_inner_recruit } }
			random_character = {
				limit = {
					is_alive = yes
					is_adult = yes
					faith = root.faith
					is_ruler = no
					is_courtier = no
					OR = {
						has_trait = good_meridians
						has_trait = excellent_meridians
						has_trait = heavenly_meridians
					}
				}
				save_scope_as = sect_exam_inner_wanderer
			}

			scope:sect_exam_inner_wanderer ?= {
				root = { add_courtier = scope:sect_exam_inner_wanderer }
				if = { limit = { NOT = { has_trait = sect_disciple } } add_trait = sect_disciple }
				remove_character_modifier = outer_court_disciple
				add_character_modifier = { modifier = inner_court_disciple days = 5475 }
				if = {
					limit = {
						NOT = {
							OR = {
								has_trait = weapon_combat_talent_1
								has_trait = weapon_combat_talent_2
								has_trait = weapon_combat_talent_3
								has_trait = weapon_combat_talent_4
								has_trait = weapon_combat_talent_5
							}
						}
					}
					xianxia_apply_weapon_talent_lottery_effect = yes
				}
			}
		}

		# Last fallback: create a new recruit directly.
		if = {
			limit = {
				NOT = { exists = scope:sect_exam_inner_recruit }
				NOT = { exists = scope:sect_exam_inner_wanderer }
			}
			create_character = {
				age = { 16 25 }
				gender_female_chance = 35
				employer = root
				faith = root.faith
				culture = root.culture
				dynasty = none
				random_traits = no
				trait = cultivation_qi_gathering
				trait = sect_disciple
				trait = good_meridians
				save_scope_as = sect_exam_inner_created
			}
			scope:sect_exam_inner_created = {
				add_character_modifier = { modifier = inner_court_disciple days = 5475 }
				xianxia_apply_weapon_talent_lottery_effect = yes
			}
		}

		# Outer-court disciple recruitment from realm/court.
			random_courtier = {
				limit = {
					is_alive = yes
					is_adult = yes
					faith = root.faith
					NOT = { is_ruler = yes }
					NOT = { has_trait = sect_elder }
				NOT = { has_character_modifier = inner_court_disciple }
				NOT = { has_character_modifier = outer_court_disciple }
			}
			save_scope_as = sect_exam_outer_recruit
		}

		scope:sect_exam_outer_recruit ?= {
			if = { limit = { NOT = { has_trait = sect_disciple } } add_trait = sect_disciple }
			add_character_modifier = { modifier = outer_court_disciple days = 3650 }
		}

		# Fallback to wandering disciple for outer court.
		if = {
			limit = { NOT = { exists = scope:sect_exam_outer_recruit } }
			random_character = {
				limit = {
					is_alive = yes
					is_adult = yes
					faith = root.faith
					is_ruler = no
					is_courtier = no
				}
				save_scope_as = sect_exam_outer_wanderer
			}
			scope:sect_exam_outer_wanderer ?= {
				root = { add_courtier = scope:sect_exam_outer_wanderer }
				if = { limit = { NOT = { has_trait = sect_disciple } } add_trait = sect_disciple }
				add_character_modifier = { modifier = outer_court_disciple days = 3650 }
				if = {
					limit = {
						NOT = {
							OR = {
								has_trait = weapon_combat_talent_1
								has_trait = weapon_combat_talent_2
								has_trait = weapon_combat_talent_3
								has_trait = weapon_combat_talent_4
								has_trait = weapon_combat_talent_5
							}
						}
					}
					xianxia_apply_weapon_talent_lottery_effect = yes
				}
			}
		}

		# Final fallback for outer-court disciple.
		if = {
			limit = {
				NOT = { exists = scope:sect_exam_outer_recruit }
				NOT = { exists = scope:sect_exam_outer_wanderer }
			}
			create_character = {
				age = { 16 30 }
				gender_female_chance = 45
				employer = root
				faith = root.faith
				culture = root.culture
				dynasty = none
				random_traits = no
				trait = cultivation_qi_gathering
				trait = sect_disciple
				save_scope_as = sect_exam_outer_created
			}
			scope:sect_exam_outer_created = {
				add_character_modifier = { modifier = outer_court_disciple days = 3650 }
				xianxia_apply_weapon_talent_lottery_effect = yes
			}
		}
	}
}

			}

		}
	}
}

							has_trait = weapon_combat_talent_3
							has_trait = weapon_combat_talent_4
							has_trait = weapon_combat_talent_5
						}
					}
				}
				xianxia_apply_weapon_talent_lottery_effect = yes
			}
			if = {
				limit = { qualifies_inner_court_acceptance = yes }
				add_character_modifier = { modifier = inner_court_disciple days = 5475 }
			}
			else = {
				add_character_modifier = { modifier = outer_court_disciple days = 3650 }
			}
			add_learning_lifestyle_xp = minor_lifestyle_xp
		}

		scope:sect_exam_candidate_2 ?= {
			root = { add_courtier = scope:sect_exam_candidate_2 }
			if = {
				limit = { NOT = { has_trait = sect_disciple } }
				add_trait = sect_disciple

				limit = {
					NOT = {

							has_trait = weapon_combat_talent_1
							has_trait = weapon_combat_talent_2
							has_trait = weapon_combat_talent_3

						}

				xianxia_apply_weapon_talent_lottery_effect = yes
			}

			else = {

			root = {

				add_character_modifier = { modifier = outer_court_disciple days = 3650 }
			}
		}

		if = {
			limit = {
				NOT = { exists = scope:sect_exam_candidate_1 }
				NOT = { exists = scope:sect_exam_candidate_2 }
			}
			create_character = {
				age = { 16 28 }
				gender_female_chance = 35
				employer = root
				faith = root.faith
				culture = root.culture
				dynasty = none
				trait = cultivation_qi_gathering
				save_scope_as = generated_exam_recruit
				random_traits = no
			}
				scope:generated_exam_recruit = {
					add_trait = sect_disciple
					xianxia_apply_weapon_talent_lottery_effect = yes
					random_list = {
					65 = { add_trait = good_meridians }
					27 = { add_trait = excellent_meridians }

				}
				if = {
					limit = { qualifies_inner_court_acceptance = yes }
